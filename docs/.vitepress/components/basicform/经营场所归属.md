
## 引入
```
<script>
    // 导入组件
    import StorePlaceOwnerRadio from './components/StorePlaceOwnerRadio.vue';

</script> 
```
## 代码演示
```
<input
    type="radio"
    :id="`${paramName}-rent`"
    :value="options[0].value"
    v-model="selectedOption"
    @change="$emit('update:initialValue', selectedOption)"
>
<input
    type="radio"
    :id="`${paramName}-own`"
    :value="options[1].value"
    v-model="selectedOption"
    @change="$emit('update:initialValue', selectedOption)"
>
```
